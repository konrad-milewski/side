function scrollPage(direction) { let scrollInterval = null; const scrollStep = 10; const scrollFn = () => { window.scrollBy(0, direction * scrollStep); }; const startScroll = () => { scrollInterval = setInterval(scrollFn, 10); }; const stopScroll = () => { clearInterval(scrollInterval); }; startScroll(); // Listen for mouseup events on the document to stop scrolling document.addEventListener('mouseup', stopScroll); // Stop scrolling when the button is released const button = event.target; button.addEventListener('mouseup', stopScroll); // Start scrolling when the button is held down button.addEventListener('mousedown', startScroll); }
